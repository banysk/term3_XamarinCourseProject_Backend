-- drop if exists
DROP TABLE IF EXISTS OPERATION_TYPES CASCADE;

CREATE TABLE OPERATION_TYPES
(id SERIAL, -- ID операции
 type_name VARCHAR(5), -- Название операции
 -----
 PRIMARY KEY(id),
 UNIQUE(type_name),
 -----
 CHECK(id IS NOT NULL),
 CHECK(type_name IS NOT NULL)
);

	type_name
